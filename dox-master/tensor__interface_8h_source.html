347  result_tensor binary(const Right& right, Op&& op, const Perm& perm) const {
348  // return result_tensor(*this, right, op, perm);

357  if constexpr (is_tot && is_bperm) {
358  if (inner_size(perm) != 0) {
359  auto inner_perm = inner(perm);
